<project_specification>
  <project_name>Todo App - Task Management Interface</project_name>

  <overview>
    Build a polished, full-featured task management web application. Users can create,
    edit, complete, and delete tasks organized into categories with color-coded labels.
    The app supports due dates, priority levels, search, drag-and-drop reordering, and
    recurring tasks. The UI should be clean, responsive, and accessible with dark/light
    theme support.
  </overview>

  <technology_stack>
    <frontend>
      <framework>React 18 with Vite</framework>
      <styling>Tailwind CSS (via CDN)</styling>
      <state_management>React hooks and context</state_management>
      <routing>React Router v6</routing>
      <drag_and_drop>@dnd-kit/core for drag-and-drop</drag_and_drop>
      <icons>Lucide React icons</icons>
      <dates>date-fns for date formatting</dates>
      <port>Launch on port 5173</port>
    </frontend>
    <backend>
      <runtime>Node.js with Express</runtime>
      <database>SQLite with better-sqlite3</database>
      <validation>Express-validator for input validation</validation>
      <port>Launch on port 3001</port>
    </backend>
    <communication>
      <api>RESTful JSON endpoints</api>
      <cors>CORS enabled for localhost development</cors>
    </communication>
  </technology_stack>

  <prerequisites>
    <environment_setup>
      - Node.js 18+ installed
      - npm as package manager

  <core_features>
    <task_management>
      - Create tasks with title (required), description (optional), due date, priority, and category
      - Edit task title and description inline with click-to-edit
      - Mark tasks as complete/incomplete with animated checkbox
      - Delete tasks with confirmation dialog ("Are you sure?")
      - Drag-and-drop reordering within and across categories
      - Batch select and bulk actions (complete, delete, move category)
      - Duplicate an existing task
      - Task detail modal with full editing capabilities
      - Subtasks/checklist within a task (add, toggle, remove sub-items)
      - Task notes/comments section in detail view
    </task_management>

    <priority_system>
      - Four priority levels: Urgent (red), High (orange), Medium (yellow), Low (green)
      - Priority badge displayed on task card
      - Sort tasks by priority
      - Filter tasks by priority level
      - Default priority: Medium
    </priority_system>

    <due_dates>
      - Date picker for setting due dates
      - Visual indicators: overdue (red), due today (orange), due this week (yellow)
      - "No date" option to clear due date
      - Sort tasks by due date
      - Filter: overdue, due today, due this week, upcoming, no date
      - Relative date display ("2 days ago", "in 3 days", "today")
    </due_dates>

    <recurring_tasks>
      - Set task recurrence: daily, weekly, monthly, custom interval
      - When completing a recurring task, auto-create next occurrence
      - Show recurrence icon and interval on task card
      - Edit or remove recurrence from task detail
    </recurring_tasks>

    <categories>
      - Create custom categories with name and color picker
      - Edit category name and color
      - Delete category (with option to reassign or delete tasks)
      - Filter tasks by category (single or multiple)
      - Default categories: Work (#3B82F6 blue), Personal (#8B5CF6 purple), Shopping (#10B981 green), Health (#EF4444 red)
      - Category sidebar with task count badges
      - "All Tasks" and "Uncategorized" virtual categories
      - Drag tasks between category sections
    </categories>

    <search_and_filter>
      - Global search bar with real-time filtering as you type
      - Search matches title and description content
      - Highlight matching text in results
      - Combined filters: category + priority + due date + status
      - Clear all filters button
      - Show active filter count badge
      - Save filter presets (e.g., "My urgent work tasks")
    </search_and_filter>

    <sorting>
      - Sort by: created date, due date, priority, alphabetical, manual order
      - Toggle ascending/descending
      - Persist sort preference per category
      - Sort indicator in column header
    </sorting>

    <statistics_dashboard>
      - Tasks completed today/this week/this month
      - Completion rate percentage with progress ring
      - Tasks by category (pie/donut chart)
      - Tasks by priority (bar chart)
      - Overdue tasks count with warning
      - Streak: consecutive days with completed tasks
    </statistics_dashboard>
  </core_features>

  <database_schema>
    <tables>
      <categories>
        - id INTEGER PRIMARY KEY AUTOINCREMENT
        - name TEXT NOT NULL
        - color TEXT NOT NULL (hex color code)
        - position INTEGER NOT NULL DEFAULT 0
        - created_at TEXT NOT NULL DEFAULT (datetime('now'))
        - updated_at TEXT NOT NULL DEFAULT (datetime('now'))
      </categories>

      <tasks>
        - id INTEGER PRIMARY KEY AUTOINCREMENT
        - title TEXT NOT NULL
        - description TEXT DEFAULT ''
        - is_completed INTEGER NOT NULL DEFAULT 0
        - priority TEXT NOT NULL DEFAULT 'medium' CHECK(priority IN ('urgent','high','medium','low'))
        - category_id INTEGER REFERENCES categories(id) ON DELETE SET NULL
        - due_date TEXT (ISO 8601 date string or NULL)
        - recurrence TEXT DEFAULT NULL (JSON: {"interval":"daily|weekly|monthly","every":1})
        - position INTEGER NOT NULL DEFAULT 0
        - created_at TEXT NOT NULL DEFAULT (datetime('now'))
        - updated_at TEXT NOT NULL DEFAULT (datetime('now'))
        - completed_at TEXT DEFAULT NULL
      </tasks>

      <subtasks>
        - id INTEGER PRIMARY KEY AUTOINCREMENT
        - task_id INTEGER NOT NULL REFERENCES tasks(id) ON DELETE CASCADE
        - title TEXT NOT NULL
        - is_completed INTEGER NOT NULL DEFAULT 0
        - position INTEGER NOT NULL DEFAULT 0
        - created_at TEXT NOT NULL DEFAULT (datetime('now'))
      </subtasks>

      <task_notes>
        - id INTEGER PRIMARY KEY AUTOINCREMENT
        - task_id INTEGER NOT NULL REFERENCES tasks(id) ON DELETE CASCADE
        - content TEXT NOT NULL
        - created_at TEXT NOT NULL DEFAULT (datetime('now'))
      </task_notes>

      <filter_presets>
        - id INTEGER PRIMARY KEY AUTOINCREMENT
        - name TEXT NOT NULL
        - filters TEXT NOT NULL (JSON: {"categories":[],"priorities":[],"due":"overdue"})
        - created_at TEXT NOT NULL DEFAULT (datetime('now'))
      </filter_presets>
    </tables>
  </database_schema>

  <api_endpoints_summary>
    <tasks>
      - GET    /api/tasks                    List tasks (query: category, priority, due, status, search, sort, order)
      - POST   /api/tasks                    Create task (body: title, description, priority, category_id, due_date, recurrence)
      - GET    /api/tasks/:id                Get task with subtasks and notes
      - PUT    /api/tasks/:id                Update task fields
      - DELETE /api/tasks/:id                Delete task
      - PATCH  /api/tasks/:id/toggle         Toggle complete/incomplete
      - POST   /api/tasks/:id/duplicate      Duplicate task
      - PUT    /api/tasks/reorder            Bulk update positions (body: [{id, position}])
      - POST   /api/tasks/bulk               Bulk action (body: {ids, action: "complete"|"delete"|"move", category_id?})
    </tasks>

    <subtasks>
      - GET    /api/tasks/:id/subtasks       List subtasks
      - POST   /api/tasks/:id/subtasks       Create subtask
      - PUT    /api/subtasks/:id             Update subtask
      - DELETE /api/subtasks/:id             Delete subtask
      - PATCH  /api/subtasks/:id/toggle      Toggle subtask complete
    </subtasks>

    <notes>
      - GET    /api/tasks/:id/notes          List notes for task
      - POST   /api/tasks/:id/notes          Add note
      - DELETE /api/notes/:id                Delete note
    </notes>

    <categories>
      - GET    /api/categories               List all categories with task counts
      - POST   /api/categories               Create category
      - PUT    /api/categories/:id           Update category
      - DELETE /api/categories/:id           Delete category (query: reassign_to?)
      - PUT    /api/categories/reorder       Reorder categories
    </categories>

    <filters>
      - GET    /api/filters                  List saved filter presets
      - POST   /api/filters                  Save filter preset
      - DELETE /api/filters/:id             Delete filter preset
    </filters>

    <stats>
      - GET    /api/stats                    Get dashboard statistics
      - GET    /api/stats/history            Get completion history (query: days=30)
    </stats>
  </api_endpoints_summary>

  <key_interactions>
    <task_creation_flow>
      1. User clicks "Add Task" or presses keyboard shortcut (N)
      2. Inline input appears at top of task list with focus
      3. User types task title, presses Enter
      4. Task created with default priority (Medium), current category, no due date
      5. Task card appears with slide-down animation
      6. User can click task to open detail modal for full editing
    </task_creation_flow>

    <task_completion_flow>
      1. User clicks checkbox on task card
      2. Checkbox fills with animated checkmark
      3. Title gets strikethrough animation
      4. After 500ms delay, task slides to "Completed" section at bottom
      5. If task has recurrence, new task auto-created for next occurrence
      6. Statistics update in real-time
    </task_completion_flow>

    <task_editing_flow>
      1. User clicks task card (not checkbox)
      2. Detail modal opens with current values
      3. All fields are editable inline
      4. Changes auto-save on field blur (debounced 500ms)
      5. "Delete" button shows confirmation dialog
      6. Close modal with X, Escape key, or click outside
    </task_editing_flow>

    <category_management_flow>
      1. User clicks "Add Category" in sidebar
      2. Inline input appears with color picker
      3. User enters name and selects color
      4. Category appears in sidebar with 0 count badge
      5. Right-click category for edit/delete context menu
      6. Deleting category prompts: reassign tasks or delete them
    </category_management_flow>

    <search_and_filter_flow>
      1. User types in search bar (or presses / shortcut)
      2. Tasks filter in real-time as user types (debounced 300ms)
      3. Matching text highlighted in yellow in results
      4. User can combine with category/priority/due filters
      5. Active filter count shown as badge on filter icon
      6. "Clear all" button resets to default view
    </search_and_filter_flow>

    <drag_and_drop_flow>
      1. User grabs drag handle on task card
      2. Task lifts with shadow + slight scale
      3. Drop zones highlight as task moves over them
      4. Drop task to reorder within list
      5. Drop task on category in sidebar to reassign
      6. Position saved to database on drop
    </drag_and_drop_flow>
  </key_interactions>
